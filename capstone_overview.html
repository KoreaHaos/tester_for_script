Pulse Signal (MSP430) :


#define LED RED_LED

void setup() {                
	// initialize the digital pin as an output.
	pinMode(LED, OUTPUT);
	// start out with the led off.
	digitalWrite(LED, LOW);  
}

void loop() {
	  
	// led off.
	digitalWrite(LED, LOW);
	// wait four seconds.
	delay(4000);

	// 4 loops for 4 pulses.
	int i;

	// pulse 1
	for(i = 0; i < 50; i++)
	{
		// turn the LED on (HIGH is the voltage level)
		digitalWrite(LED, HIGH);
		// wait for 2 ms
		delay(2);
		// turn the LED off by making the voltage LOW
		digitalWrite(LED, LOW);
		// wait for 19 ms
		delay(19);
	}
	// off for half a second.
	digitalWrite(LED, LOW);
	delay(500);

	// pulse 2
	for(i = 0; i < 50; i++)
	{
		digitalWrite(LED, HIGH);
		delay(7);
		digitalWrite(LED, LOW);
		delay(13);
	}
	// off for half a second.
	digitalWrite(LED, LOW);
	delay(500);
	
	
	// pulse 3
	for(i = 0; i < 50; i++)
	{
		digitalWrite(LED, HIGH);
		delay(10);
		digitalWrite(LED, LOW);
		delay(10);
	}
	// off for half a second.
	digitalWrite(LED, LOW);
	delay(500);

	// pulse 4
	for(i = 0; i < 50; i++)
	{
		digitalWrite(LED, HIGH);
		delay(19);
		digitalWrite(LED, LOW);
		delay(1);
	}
	// off for half a second.
	digitalWrite(LED, LOW);
	delay(500);
	}
}
